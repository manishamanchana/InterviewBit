void reverse(vector<int>&v)
{
    int sz=v.size();
    int i=0,j=sz-1;
    while(i<j)
    {
        swap(v[i],v[j]);
        i++;
        j--;
    }
}
void Solution::rotate(vector<vector<int> > &A)
{
    int n=A.size();
    int m=A[0].size();
    for(int i=0;i<n;i++)
    {
        for(int j=i;j<m;j++)
        {
            swap(A[i][j],A[j][i]);
        }
    }
    for(int i=0;i<n;i++)
    {
        reverse(A[i]);
    }
}
